w = 2*pi*f

v_em = 1 / sqrt(e0 * u0) = 299 792 458 m/s  
v_prop = f * lambda

Z0 = sqrt( (R + jwL) / (G + jwC) )  
p = (Z - Z0) / (Z + Z0)
